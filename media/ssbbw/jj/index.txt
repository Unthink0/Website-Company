<script>
const grid = document.getElementById("image-grid");
const sortSelect = document.getElementById("sortSelect");

// Save initial order
Array.from(grid.children).forEach((tile, index)=>{
  tile.dataset.originalOrder = index;
});

// Update the video counter
function updateVideoCounter() {
  const totalVideos = document.querySelectorAll('#image-grid .tile').length;
  document.getElementById("counter-value").textContent = totalVideos;
}

// Run on page load
window.addEventListener("DOMContentLoaded", () => {
  sortSelect.value = "default";
  sortGrid();
  updateVideoCounter(); // update the counter after sorting
});

const searchInput = document.getElementById("searchInput");

searchInput.addEventListener("input", () => {
  const query = searchInput.value.toLowerCase();
  const tiles = document.querySelectorAll('#image-grid .tile');

  tiles.forEach(tile => {
    const caption = tile.querySelector('.caption').textContent.toLowerCase();
    tile.style.display = caption.includes(query) ? "block" : "none";
  });

  updateVideoCounter(); // update the counter to reflect search results
});


// Sorting function
function sortGrid(){
  const tiles = Array.from(grid.children);
  let sorted;
  if(sortSelect.value === "asc"){
    sorted = tiles.sort((a,b)=> parseInt(a.dataset.value) - parseInt(b.dataset.value));
  } else if(sortSelect.value === "desc"){
    sorted = tiles.sort((a,b)=> parseInt(b.dataset.value) - parseInt(a.dataset.value));
  } else {
    sorted = tiles.sort((a,b)=> a.dataset.originalOrder - b.dataset.originalOrder);
  }

  grid.innerHTML = "";
  sorted.forEach(tile => grid.appendChild(tile));
}

// Trigger sorting on dropdown change
sortSelect.addEventListener("change", sortGrid);

// Default sort: use original order
window.addEventListener("DOMContentLoaded", ()=> {
  sortSelect.value = "default";
  sortGrid();
});
</script>
